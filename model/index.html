Create GUID

Format của id aaaaaaaa-bbbb-cccc-dddd-eeeeeeeeeeee

$id = Sugarcrm\Sugarcrm\Util\Uuid::uuid1();

Create new record

//Create bean
$bean = BeanFactory::newBean($module);

//Populate bean fields
$bean->name = 'Example Record';

//Save
$bean->save();

//Retrieve the bean id
$record_id = $bean->id;

Saving a Bean with a Specific ID

//Create bean
$bean = BeanFactory::newBean($module);

//set the new flag
$bean->new_with_id = true;

//Set the record id with a static id
$id = '38c90c70-7788-13a2-668d-513e2b8df5e1';
$bean->id = $id;

//or have the system generate an id for you
//$id = Sugarcrm\Sugarcrm\Util\Uuid::uuid1()
//$bean->id = $id;

//Populate bean fields
$bean->name = 'Example Record';

//Save
$bean->save();

Check a Bean record is new or existing

if (!isset($bean->fetched_row['id'])) {
    //new record
} else {
    //existing record
}

Lưu ý: Nếu code xử lý trong logichook before_save hoặc after_save thì check như sau:

function hook_method($bean, $event, $arguments)
    {
        if (isset($arguments['isUpdate']) && $arguments['isUpdate'] == false) {
            //new record
        } else {
            //existing record
        }
    }

Retrieve and update a bean record

//Retrieve bean
$bean = BeanFactory::retrieveBean($module, $id);

//Fields to update
$bean->name = 'Updated Name';

//Save
$bean->save();
  • Trong 1 số trường hợp cần bypass security thì cú pháp retrieveBean như sau:
$bean = BeanFactory::retrieveBean(
    $module, 
    $record_id, 
    array(
        'disable_row_level_security' => true
    )
);
  • Update bean nhưng không thay đổi giá trị date_modified thì set modified flag như sau:
$bean->update_date_modified = false;

Delete a Bean

//Retrieve bean
$bean = BeanFactory::retrieveBean($module, $id);

//Set deleted to true
$bean->mark_deleted($id);

//Save
$bean->save();

Fetching Relationships

//If relationship is loaded
if ($bean->load_relationship($link)) {
    //Fetch related beans
    $relatedBeans = $bean->$link->getBeans();

    //Fetch related record IDs
    $relatedBeanID = $bean->$link->get();
}

Add relationsihps

//If relationship is loaded
if ($bean->load_relationship($link)) {
    $bean->$link->add('related-Bean-ID');
}
